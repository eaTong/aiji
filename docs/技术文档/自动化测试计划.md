### 3.3 E2E 测试用例

```typescript
// e2e/training.spec.ts

test.describe('快捷按钮推荐流程', () => {
  test('用户点击「练什么」应展示推荐卡片', async ({ page }) => {
    await page.goto('/pages/index/index')
    await page.click('text=练什么')
    // 验证推荐卡片出现
    await expect(page.locator('.recommendation-card')).toBeVisible()
  })

  test('无目标用户应被引导设置目标', async ({ page }) => {
    // 创建无目标用户
    await page.goto('/pages/index/index')
    await page.click('text=练什么')
    // 验证目标选择弹窗
    await expect(page.locator('.modal-content')).toBeVisible()
  })
})

test.describe('训练记录保存流程', () => {
  test('应能选择动作并保存记录', async ({ page }) => {
    await page.goto('/pages/training/today')
    await page.click('text=开始训练')
    await page.click('text=添加动作')
    // 选择动作
    await page.click('text=胸部')
    await page.click('text=杠铃卧推')
    // 填写重量
    await page.fill('input[placeholder="0"]', '60')
    // 保存
    await page.click('text=保存记录')
    // 验证成功提示
    await expect(page.locator('text=已记录')).toBeVisible()
  })
})
```

## 四、CI/CD 集成

### 4.1 GitHub Actions 配置

```yaml
# .github/workflows/test.yml
name: Test

on: [push, pull_request]

jobs:
  backend-test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Run backend tests
        run: |
          cd backend
          npm install
          npm test

  frontend-test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Run frontend tests
        run: |
          cd front
          npm install
          npm run test:e2e
```
