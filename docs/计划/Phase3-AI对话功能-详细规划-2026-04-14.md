## 四、追问澄清模块（Clarification Engine）

### 4.0 概述

追问澄清是 AI 对话系统的关键能力，用于处理用户输入信息不完整或意图模糊的情况。

```
用户输入 → 意图识别 → 信息完整性检查 → [信息完整?] → 生成回复
                                          ↓
                                    [信息不足?] → 追问澄清引擎 → 追问话术
                                          ↓
                                    [意图模糊?] → 意图消歧 → 选项卡片
```

```typescript
// intentChecker.ts
interface IntentCheckResult {
  shouldClarify: boolean
  reason: ClarifyReason
  missingFields: Field[]
  suggestedQuestion?: string
}

type ClarifyReason =
  | 'MISSING_NUMBER'        // 缺少数字
  | 'MISSING_UNIT'          // 缺少单位
  | 'MISSING_DATE'          // 缺少日期
  | 'MISSING_MUSCLE'        // 缺少肌群
  | 'MISSING_EXERCISE'      // 缺少动作
  | 'MISSING_PLAN'          // 缺少计划引用
  | 'AMBIGUOUS_INTENT'      // 意图模糊
  | 'MULTIPLE_INTENTS'      // 多意图
  | 'INCOMPLETE_SENTENCE'   // 句子不完整
  | 'CONTRADICTORY_INFO'    // 信息矛盾
  | 'INVALID_INPUT'         // 无效输入（格式错误）⭐ 新增
  | 'PARTIAL_CONFIRMATION'  // 部分信息待确认 ⭐ 新增
  | 'DEFER_LATER'          // 稍后再说 ⭐ 新增

// 核心检查函数
export async function checkIntentCompleteness(
  intent: Intent,
  extractedEntities: Record<string, any>
): Promise<IntentCheckResult> {
  const requiredFields = INTENT_REQUIREMENTS[intent.type]
  const missing: Field[] = []

  for (const field of requiredFields) {
    if (!extractedEntities[field.name] && !hasDefaultValue(field)) {
      missing.push(field)
    }
  }

  if (missing.length > 0) {
    return {
      shouldClarify: true,
      reason: determineClarifyReason(missing),
      missingFields: missing,
      suggestedQuestion: generateClarifyQuestion(missing)
    }
  }

  return { shouldClarify: false, reason: null, missingFields: [] }
}
```

### 4.4 意图消歧逻辑

#### 4.4.1 多意图处理

```typescript
// intentDisambiguation.ts
interface DisambiguationResult {
  primaryIntent: Intent
  secondaryIntents: Intent[]
  clarificationNeeded: boolean
  suggestion?: string
}

// 多意图检测与排序
export function disambiguateIntents(
  intents: Intent[],
  context: ChatContext
): DisambiguationResult {
  // 按优先级排序
  const sorted = intents.sort((a, b) => {
    const priorityA = INTENT_PRIORITY[a.type]
    const priorityB = INTENT_PRIORITY[b.type]
    return priorityA - priorityB
  })

  const [primary, ...secondary] = sorted

  // 如果多个意图优先级相同，需要澄清
  const hasConflict = sorted.slice(1).some(
    i => INTENT_PRIORITY[i.type] === INTENT_PRIORITY[primary.type]
  )

  if (hasConflict) {
    return {
      primaryIntent: primary,
      secondaryIntents: secondary,
      clarificationNeeded: true,
      suggestion: `我理解你想做 ${primary.type} 和 ${secondary[0].type}，想先做哪个？`
    }
  }

  return {
    primaryIntent: primary,
    secondaryIntents: secondary,
    clarificationNeeded: false
  }
}

// 意图优先级
const INTENT_PRIORITY: Record<IntentType, number> = {
  RECORD_WEIGHT: 1,
  RECORD_MEASUREMENT: 1,
  RECORD_TRAINING: 1,
  DIET_RECORD: 1,
  TRAINING_RECOMMEND: 2,
  QUERY_RECOVERY: 3,
  QUERY_PR: 3,
  QUERY_TREND: 3,
  GREETING: 10,
  CHITCHAT: 10
}
```

```typescript
interface FuzzyMatch {
  intent: IntentType
  confidence: number
  matchedPattern: string
}

// 模糊匹配规则
const FUZZY_PATTERNS: Record<IntentType, string[]> = {
  RECORD_WEIGHT: [
    /体重/i,
    /kg/i,
    /公斤/i,
    /斤/i,
    /磅/i,
    /重/i
  ],

  TRAINING_RECOMMEND: [
    /练什么/i,
    /训练/i,
    /动/i,
    /练/i,
    /健身/i,
    /运动/i
  ],

  RECORD_TRAINING: [
    /记录.*训练/i,
    /练.*记/i,
    /做完/i,
    /练完/i,
    /训练.*录/i
  ],

  QUERY_RECOVERY: [
    /恢复/i,
    /肌肉.*状态/i,
    /酸/i,
    /休息/i,
    /疲劳/i
  ]
}

// 模糊匹配函数
export function fuzzyMatchIntent(message: string): FuzzyMatch[] {
  const results: FuzzyMatch[] = []

  for (const [intent, patterns] of Object.entries(FUZZY_PATTERNS)) {
    for (const pattern of patterns) {
      if (pattern.test(message)) {
        const existing = results.find(r => r.intent === intent)
        if (existing) {
          existing.confidence += 0.2
        } else {
          results.push({
            intent: intent as IntentType,
            confidence: 0.3,
            matchedPattern: pattern.source
          })
        }
      }
    }
  }

  return results.sort((a, b) => b.confidence - a.confidence)
}
```

```typescript
    }
  })

  sessionId: string,
  record: ClarificationRecord
): Promise<void> {
  const session = await prisma.chatSession.findUnique({
    where: { id: sessionId }
  })

  const history = session?.context?.clarificationHistory || []
  history.push(record)

  await prisma.chatSession.update({
    where: { id: sessionId },
    data: {
      context: {
        ...session.context,
        clarificationHistory: history,
        retryCount: (session.context?.retryCount || 0) + 1
      }
    }
  })
}
```

#### 4.5.2 上下文延续处理

```typescript
// contextContinuator.ts
interface ContinuationResult {
  shouldContinue: boolean
  previousContext?: ClarificationContext
  mergedEntities: Record<string, any>
}

// 检查是否应该延续上一轮上下文
export async function checkContextContinuation(
  userId: string,
  newMessage: string
): Promise<ContinuationResult> {
  const session = await getOrCreateSession(userId)

  // 检查是否在追问状态
  if (session.status !== 'AWAITING_CLARIFICATION') {
    return { shouldContinue: false, mergedEntities: {} }
  }

  const context: ClarificationContext = session.context

  // 检查是否是追问回复（而非切换话题）
  const isContinuation = detectContinuation(newMessage, context)

  if (!isContinuation) {
    // 用户切换话题，清除上下文
    await clearSession(session.id)
    return { shouldContinue: false, mergedEntities: {} }
  }

  return {
    shouldContinue: true,
    previousContext: context,
    mergedEntities: {
      ...context.extractedEntities
    }
  }
}

// 检测是否是上下文延续
function detectContinuation(
  message: string,
  context: ClarificationContext
): boolean {
  const continuationIndicators = [
    /^\d+[\d.]*$/,           // 纯数字
    /^(kg|斤|磅|cm|英寸)$/i,  // 单位
    /^(今天|昨天|前天|上周)$/, // 时间词
    /^(胸|背|腿|肩|臂|腹)$/,   // 肌群简称
    /^(是|对|嗯|ok|好的)$/i,   // 肯定回复
    /^(不|没|别)/i            // 否定回复
  ]

  // 如果消息匹配延续指示符，认为是延续
  for (const pattern of continuationIndicators) {
    if (pattern.test(message.trim())) {
      return true
    }
  }

  // 如果距离上次交互超过 2 分钟，认为是切换
  const timeSinceLastUpdate = Date.now() - new Date(context.updatedAt).getTime()
  if (timeSinceLastUpdate > 120000) {
    return false
  }

  return true
}
```

### 4.6 追问流程实现

#### 4.6.1 追问服务

```typescript
// clarificationService.ts
export class ClarificationService {
  constructor(
    private prisma: PrismaClient,
    private intentChecker: IntentChecker,
    private templateEngine: TemplateEngine,
    private sessionManager: SessionManager
  ) {}

  // 处理用户消息的主流程
  async processMessage(
    userId: string,
    message: string
  ): Promise<AIResponse> {
    // 1. 检查是否在追问状态
    const continuation = await checkContextContinuation(userId, message)

    let intent: Intent
    let extractedEntities: Record<string, any>

    if (continuation.shouldContinue && continuation.previousContext) {
      // 2a. 延续上下文：提取本轮信息并合并
      const currentIntent = parseIntent(message)
      const currentEntities = extractEntities(message, currentIntent.type)
      extractedEntities = {
        ...continuation.previousContext.extractedEntities,
        ...currentEntities
      }
      intent = continuation.previousContext.pendingIntent
    } else {
      // 2b. 新对话：正常解析
      intent = parseIntent(message)
      extractedEntities = extractEntities(message, intent.type)
    }

    // 3. 检查信息完整性
    const checkResult = await this.intentChecker.checkIntentCompleteness(
      intent,
      extractedEntities
    )

    if (checkResult.shouldClarify) {
      // 4. 需要追问
      const session = await this.sessionManager.getOrCreateSession(userId)

      // 生成追问话术
      const question = this.templateEngine.generateQuestion(
        checkResult.missingFields,
        continuation.previousContext
      )

      // 更新会话状态
      await this.sessionManager.updateContext(session.id, {
        pendingIntent: intent,
        extractedEntities,
        missingFields: checkResult.missingFields
      })

      await this.prisma.chatSession.update({
        where: { id: session.id },
        data: { status: 'AWAITING_CLARIFICATION' }
      })

      return {
        type: 'text',
        content: question,
        sessionId: session.id
      }
    }

    // 5. 信息完整，执行意图
    await this.sessionManager.completeSession(
      (await this.sessionManager.getOrCreateSession(userId)).id
    )

    return await this.executeIntent(intent, extractedEntities)
  }
}
```

```typescript
async function handleExceededLimit(
  sessionId: string,
  pendingIntent: Intent,
  exceededField: Field
): Promise<AIResponse> {
  const strategy = determineExitStrategy(pendingIntent, exceededField)

  switch (strategy) {
    case 'SKIP_FIELD':
      // 跳过该字段，继续处理其他信息
      return await executeWithSkippedField(sessionId, exceededField)

    case 'USE_DEFAULTS':
      // 使用默认值
      const defaultValue = getDefaultValue(exceededField)
      return await executeWithDefault(sessionId, exceededField, defaultValue)

    case 'CANCEL_INTENT':
      // 取消意图，友好提示
      return {
        type: 'text',
        content: `好的，暂时跳过这个信息。你想试试...吗？`
      }

    case 'OFFER_OPTIONS':
      // 提供预设选项
      return {
        type: 'card',
        cardType: 'option-choice',
        data: {
          question: `请选择一个选项：`,
          options: generateOptionsForField(exceededField)
        }
      }
  }
}

// 根据意图和字段类型决定退出策略
function determineExitStrategy(
  intent: Intent,
  field: Field
): ExitStrategy {
  // 权重相关优先使用默认值
  if (field.name === 'weight' || field.name === 'reps') {
    return 'USE_DEFAULTS'
  }

  // 动作/肌群优先提供选项
  if (field.type === 'select') {
    return 'OFFER_OPTIONS'
  }

  // 其他默认跳过
  return 'SKIP_FIELD'
}
```

### 4.8 交互冲突处理

#### 4.8.1 快捷按钮与追问冲突

当用户在追问状态下点击快捷按钮时：

```typescript
// interactionConflictHandler.ts

// 冲突处理策略
type QuickActionConflictStrategy =
  | 'INTERRUPT_ASK'   // 快捷按钮打断追问，直接执行
  | 'QUEUE_ACTION'    // 排队，等追问完成后执行
  | 'CONFIRM_ABORT'   // 确认是否放弃当前追问

interface ConflictResult {
  shouldProceed: boolean
  pendingContext?: ClarificationContext
  queuedAction?: QuickAction
}

// 检查冲突
async function handleQuickActionConflict(
  userId: string,
  quickAction: QuickAction
): Promise<ConflictResult> {
  const session = await getActiveSession(userId)

  // 没有追问在进行，直接执行
  if (!session || session.status !== 'AWAITING_CLARIFICATION') {
    return { shouldProceed: true }
  }

  // 有追问在进行
  const pendingContext = session.context

  // 如果快捷按钮与追问意图相关，询问是否打断
  if (isRelatedAction(quickAction, pendingContext.pendingIntent)) {
    return {
      shouldProceed: false,
      pendingContext,
      // 需要用户确认
    }
  }

  // 不相关的快捷按钮，直接打断追问
  await clearSession(session.id)
  return { shouldProceed: true }
}

// 判断是否相关
function isRelatedAction(
  quickAction: QuickAction,
  pendingIntent: Intent
): boolean {
  const relatedPairs: Record<QuickAction, Intent[]> = {
    'RECORD_WEIGHT': ['RECORD_WEIGHT', 'QUERY_WEIGHT'],
    'RECORD_TRAINING': ['RECORD_TRAINING', 'TRAINING_RECOMMEND'],
  }

  return relatedPairs[quickAction]?.includes(pendingIntent) ?? false
}
```

**默认策略建议**：快捷按钮**直接打断**追问状态，因为快捷按钮通常是明确的高优先级操作。

```typescript
type PushInsertStrategy =
  | 'PREPEND'        // 插入到消息顶部
  | 'APPEND'         // 追加到消息底部
  | 'HIDE_TEMP'      // 临时隐藏追问，等待用户回复后再显示
  | 'INTERLEAVE'     // 在追问之后插入

interface PushConflictResult {
  pushInserted: boolean
  clarificationPaused: boolean
}

// 处理推送冲突
async function handlePushConflict(
  userId: string,
  push: PushQueue
): Promise<PushConflictResult> {
  const session = await getActiveSession(userId)

  // 没有追问在进行，直接插入推送
  if (!session || session.status !== 'AWAITING_CLARIFICATION') {
    await insertPushToMessages(push)
    return { pushInserted: true, clarificationPaused: false }
  }

  // 有追问在进行
  // 推送优先级高于追问（如紧急预警）
  if (isHighPriorityPush(push)) {
    await insertPushToMessages(push)
    // 暂停追问，但不清除上下文
    await pauseClarification(session.id)
    return { pushInserted: true, clarificationPaused: true }
  }

  // 普通推送，追问优先
  return { pushInserted: false, clarificationPaused: false }
}

// 高优先级推送判断
function isHighPriorityPush(push: PushQueue): boolean {
  const HIGH_PRIORITY_TYPES = [
    'overtraining-warning',
    'personal-record',
    'achievement'
  ]
  return HIGH_PRIORITY_TYPES.includes(push.cardType)
}
```

**默认策略建议**：
- 紧急预警/PR/成就：**打断追问**，优先展示
- 普通推送（日报、周报）：**追问优先**，推送暂存

#### 4.8.3 多轮追问状态管理

```typescript
// multiTurnClarificationManager.ts

interface ClarificationTurn {
  turnNumber: number
  question: string
  userAnswer: string
  extractedValue: any
  timestamp: Date
}

// 记录追问轮次
async function recordClarificationTurn(
  sessionId: string,
  turn: ClarificationTurn
): Promise<void> {
  const session = await prisma.chatSession.findUnique({
    where: { id: sessionId }
  })

  const turns = session?.context?.clarificationTurns || []
  turns.push(turn)

  // 更新 Session
  await prisma.chatSession.update({
    where: { id: sessionId },
    data: {
      context: {
        ...session.context,
        clarificationTurns: turns,
        retryCount: turns.length
      }
    }
  })

  // 检查是否超过限制
  if (turns.length >= CLARIFICATION_LIMITS.MAX_TOTAL_RETRIES) {
    await handleExceededLimit(sessionId, session.context.pendingIntent, null)
  }
}
```

```typescript
// 数据迁移兼容
interface DataMigration {
  fromVersion: string
  toVersion: string
  migrate: (data: any) => any
}

// 体重记录格式迁移
const WEIGHT_MIGRATION: DataMigration = {
  fromVersion: '1.0',
  toVersion: '2.0',
  migrate: (data) => {
    // v1.0: { value: "70kg" }
    // v2.0: { value: 70, unit: "kg" }
    const match = data.value.match(/^(\d+\.?\d*)\s*(kg|斤)$/)
    if (match) {
      return {
        value: parseFloat(match[1]),
        unit: match[2],
        migrated: true
      }
    }
    return { ...data, migrated: false }
  }
}
```
